git mv nombre.del.archivo carpeta.destino/nombre.nuevo.archivo

	ej: git mv main.css css/main.css (mismo nombre)
	    git mv main.css css/m.css (cambio de nombre)

git rm (eliminar)

git checkout (+ codigo de la confirmacion)
git reset --soft (+ nombre de la confirmacion) (no elimina cambios)
git reset --hard (elimina confrimacion y cambios)
